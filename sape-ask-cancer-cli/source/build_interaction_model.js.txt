const {
  ControlInteractionModelGenerator
} = require('ask-sdk-controls'); // Phrase slots
// https://developer.amazon.com/blogs/alexa/post/a2716002-0f50-4587-b038-31ce631c0c07/enhance-speech-recognition-of-your-alexa-skills-with-phrase-slots-and-amazon-searchquery


new ControlInteractionModelGenerator().withInvocationName('sapient cancer dictionary').addIntent({
  name: 'AMAZON.StopIntent'
}).addIntent({
  name: 'AMAZON.NavigateHomeIntent'
}).addIntent({
  name: 'AMAZON.HelpIntent'
}).addIntent({
  name: 'AMAZON.CancelIntent'
}).addIntent({
  name: 'DictionaryAPIIntent',
  slots: [{
    name: 'Phrase',
    type: 'AMAZON.SearchQuery'
  }],
  samples: ['give me the definition of {Phrase}','tell me the definition of {Phrase}', 'tell me the meaning of {Phrase}','what is a {Phrase}', 'what is the meaning of {Phrase}', 'what is {Phrase}', 'define {Phrase}', 'to search for {Phrase}', 'what {Phrase} means', 'what {Phrase} are', 'what are {Phrase}', 'what {Phrase} is', 'what does {Phrase} mean', 'what does {Phrase} stand for']
})// Build and write (this overwrites the existing model)
.buildAndWrite('../skill-package/interactionModels/custom/en-US.json');
